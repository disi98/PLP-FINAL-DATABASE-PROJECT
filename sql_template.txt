username (pk), varchar(15), not null
firstName, varchar(50), not null
secondName, varchar(50), null
otherName, varchar(50), null

author
id auto_increament with user_ prefix eg. author_01, author_02, author_12: NOT NULL, unique
username (pk),varchar(15), not null
user (fk/user, username) not null
firstName, varchar(50), not null
secondName, varchar(50), null
otherName, varchar(50), null
nickName, varchar(50), null

category

hashtag

name varchar(20), not null
category (fk/category, name) not null
